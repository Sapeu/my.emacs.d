** 定义打开emacs配置文件方法
#+BEGIN_SRC emacs-lisp
(defun open-my-init-file()
  (interactive)
  (find-file "~/.emacs.d/init.el"))
#+END_SRC

** 配置load-path
#+BEGIN_SRC emacs-lisp
(add-to-list 'load-path "~/.emacs.d/lisp/")
#+END_SRC

** 配置require文件
#+BEGIN_SRC emacs-lisp
(require 'init-packages)
(require 'init-ui)
(require 'init-better-defaults)
(require 'init-org)
(require 'init-keybindings)
#+END_SRC
